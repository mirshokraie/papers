	function setup() {
		setDim(10);
		$('#startCell').click(start);

	function setDim(dimension) {
		var dim=($('#endCell').width() + $('#endCell').height()))/dimension;
		currentDim += dim;
		$('#endCell').css('height', dim+'px');
		return dim;
	}
	
	function start() {
		if(currentDim > 40)
		  $(this).css('height', currentDim+'px');
		else $(this).remove(); 
	}
