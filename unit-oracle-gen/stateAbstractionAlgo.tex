\IncMargin{0em}
\begin{algorithm}[t]
{\scriptsize
\SetKwInOut{Input}{input}\SetKwInOut{Output}{output}
\Input{The set of function states $st_i \in ST_f$ for a given function $f$, The set of relevant statements $\textsc{RelvStms}[st_i]$ for function state $st_i$ obtained from the backward slicing step}
\Output{The obtained abstracted states set $absStates$}
\BlankLine

\Begin {
\nl  $absStates \leftarrow \emptyset$\\ 
\nl \For{$st_i \in ST_f$}{
\nl  \For{$stm[st_i] \in \textsc{RelvStms}[st_i]$}{
\nl   \If{$stm[st_i] \in \textsc{RelvStms}[st]_{j=i+1}^n$}{
\nl    \If{$\textsc{WrVarType}(stm[st_i])\neq \textsc{WrVarType}(stm[st_j])$ $||$ $\textsc{DOMProps}(stm[st_i])\neq \textsc{DOMProps}(stm[st_j])$}{
\nl     $absStates \leftarrow absStates \cup st_i$\\
        }	
       }
\nl   \Else{
\nl    $absStates \leftarrow absStates \cup st_i$\\
       }   
      }
    }
\nl return $absStates$
}

\caption{Function State Abstraction} \label{Alg:stateAbstractionAlgo}
}
\end{algorithm}
